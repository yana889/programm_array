#include <iostream>
#include <Windows.h>
#include <cstdlib>

using namespace std;

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	int n, m;
	cout << "Введите кол-во строк массива: ";
	cin >> n;
	cout << "Введите кол-во столбцов массива: ";
	cin >> m;
	try {
		if (n < 2 && m < 2) {
			throw "Размерность массива должна быть >=2";
		}
	}
	catch (const char* s) {
		cout << s << endl;
	}
	int sum = 0;
	srand(time(0));
	int** a = new int* [n];
	for (int i = 0; i < n; i++)
	{
		a[i] = new int[m];
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			a[i][j] = rand() % 100;
		}
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			cout << a[i][j] << "\t";
		}
		cout << endl;
	}
	
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (j == i) {
				sum += a[i][j];
			}
		}
	}
	cout << "Сумма чисел по диагонали = " << sum << endl;
}
